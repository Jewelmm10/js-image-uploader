//upload single image function $(".input class").on("change", function(){ var PreviewID = ".preview1"; readURL(this, PreviewID); }); function readURL(input) { if (input.files && input.files[0]) { var reader = new FileReader(); reader.onload = function(e) { $('.profile-preview').css('background-image', 'url('+e.target.result +')'); } reader.readAsDataURL(input.files[0]); } }
